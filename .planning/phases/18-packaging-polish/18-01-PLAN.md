```javascript
// /tmp/verify-pack.mjs (transient one-shot — does NOT get committed; use scripts/verify-tarball.mjs in Plan 02 for the persistent version)
import { spawnSync } from 'node:child_process';

const npmCmd = process.platform === 'win32' ? 'npm.cmd' : 'npm';
const REQUIRED = ['package.json', 'README.md', 'dist/index.js', 'dist/index.d.ts', 'dist/index.js.map'];
const FORBIDDEN_EXACT = ['tsconfig.json', 'tsup.config.ts', 'vitest.config.ts'];
const FORBIDDEN_PREFIX = ['src/', 'test/'];

let allOk = true;
const evidence = [];

for (const pkg of ['@localground/mcp', '@localground/cli']) {
  const r = spawnSync(npmCmd, ['pack', '--dry-run', '--json', '-w', pkg], { encoding: 'utf8' });
  if (r.status !== 0) { console.error(`pack failed for ${pkg}:`, r.stderr); process.exit(1); }
  const parsed = JSON.parse(r.stdout);
  const files = parsed[0].files.map((f) => f.path);

  const missing = REQUIRED.filter((p) => !files.includes(p));
  const stowaways = files.filter((p) => FORBIDDEN_EXACT.includes(p) || FORBIDDEN_PREFIX.some((pre) => p.startsWith(pre)));

  evidence.push(`## ${pkg}\n\nFiles in tarball:\n${files.map((p) => `- ${p}`).join('\n')}\n\nRequired present: ${missing.length === 0 ? 'YES' : 'NO — missing: ' + missing.join(', ')}\nStowaways present: ${stowaways.length === 0 ? 'NONE' : stowaways.join(', ')}\n`);

  if (missing.length > 0) { console.error(`${pkg}: MISSING required files:`, missing); allOk = false; }
  if (stowaways.length > 0) { console.error(`${pkg}: FORBIDDEN files leaked into tarball:`, stowaways); allOk = false; }
}

const fs = await import('node:fs/promises');
await fs.writeFile('.planning/phases/18-packaging-polish/18-01-pack-evidence.txt', evidence.join('\n---\n\n'));

if (!allOk) process.exit(1);
console.log('OK — both tarballs include required files and exclude forbidden ones');
```

Run it: `node /tmp/verify-pack.mjs`. MUST exit 0. If non-zero, the `files: ["dist"]` change is wrong — STOP, do not proceed to Plan 02; report the discrepancy.

Note: the persistent / CI-wired verification is delivered by Plan 02 (`scripts/verify-tarball.mjs`). This task is the immediate proof that PKG-02 is satisfied at the manifest level.
